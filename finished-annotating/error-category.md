<script>
console.log('Dynamic Error Category - Script loaded');

class DynamicErrorCategoryViewer {
    constructor() {
        this.annotations = {};
        this.csvData = [];
        this.currentCategory = null;
        
        // Define the 7 error categories with their properties
        this.errorCategories = [
            {
                number: 1,
                name: 'Incorrect Argument Count',
                description: 'Errors related to functions being called with the wrong number of arguments',
                color: '#e74c3c',
                colorSecondary: '#c0392b'
            },
            {
                number: 2,
                name: 'Argument Unpacking Error',
                description: 'Errors related to incorrect unpacking of function arguments',
                color: '#3498db',
                colorSecondary: '#2980b9'
            },
            {
                number: 3,
                name: 'Incorrect Formula Application',
                description: 'Errors related to applying the wrong mathematical or logical formulas',
                color: '#2ecc71',
                colorSecondary: '#27ae60'
            },
            {
                number: 4,
                name: 'Off By One Error',
                description: 'Errors related to indexing or counting that are off by one',
                color: '#f39c12',
                colorSecondary: '#e67e22'
            },
            {
                number: 5,
                name: 'Incorrect Output Format',
                description: 'Errors related to returning data in the wrong format',
                color: '#9b59b6',
                colorSecondary: '#8e44ad'
            },
            {
                number: 6,
                name: 'Edge Case Handling Failure',
                description: 'Errors related to failing to handle edge cases properly',
                color: '#1abc9c',
                colorSecondary: '#16a085'
            },
            {
                number: 7,
                name: 'Syntax Error',
                description: 'Errors related to Python syntax mistakes',
                color: '#34495e',
                colorSecondary: '#2c3e50'
            }
        ];
        
        this.initializePage();
    }
    
    initializePage() {
        // Get category number from URL
        const categoryNumber = this.getCategoryNumberFromURL();
        
        if (categoryNumber < 1 || categoryNumber > 7) {
            this.showInvalidCategory();
            return;
        }
        
        this.currentCategory = this.errorCategories[categoryNumber - 1];
        
        // Load data and then display
        this.loadData().then(() => {
            this.generateCategoryNavigation();
            this.generateCategoryHeader();
            this.displayErrors();
            this.showContent();
        });
    }
    
    getCategoryNumberFromURL() {
        // Extract category number from URL path
        const pathParts = window.location.pathname.split('/');
        const categoryIndex = pathParts.findIndex(part => part === 'error-category');
        
        if (categoryIndex !== -1 && pathParts[categoryIndex + 1]) {
            return parseInt(pathParts[categoryIndex + 1]);
        }
        
        // Fallback: check URL parameters
        const urlParams = new URLSearchParams(window.location.search);
        return parseInt(urlParams.get('category')) || 1;
    }
    
    async loadData() {
        try {
            // Load data from localStorage
            const storedAnnotations = localStorage.getItem('errorAnnotations');
            const storedCsvData = localStorage.getItem('csvData');
            
            if (storedAnnotations) {
                this.annotations = JSON.parse(storedAnnotations);
            }
            
            if (storedCsvData) {
                this.csvData = JSON.parse(storedCsvData);
            }
        } catch (error) {
            console.error('Error loading category data:', error);
        }
    }
    
    generateCategoryNavigation() {
        const navContainer = document.getElementById('categoryNavigation');
        let navHTML = '';
        
        this.errorCategories.forEach(category => {
            const isActive = category.number === this.currentCategory.number;
            const activeClass = isActive ? 'active' : '';
            
            navHTML += `
                <a href="/finished-annotating/error-category/${category.number}/" 
                   class="nav-category-btn ${activeClass}"
                   style="border-color: ${category.color}; ${isActive ? `background: ${category.color}; color: white;` : `color: ${category.color};`}">
                    ${category.number}
                </a>
            `;
        });
        
        navContainer.innerHTML = navHTML;
    }
    
    generateCategoryHeader() {
        const headerContainer = document.getElementById('categoryHeader');
        const category = this.currentCategory;
        
        headerContainer.style.background = `linear-gradient(135deg, ${category.color} 0%, ${category.colorSecondary} 100%)`;
        
        headerContainer.innerHTML = `
            <h1>üìä Error Category ${category.number}</h1>
            <p><strong>${category.name}</strong></p>
            <p>${category.description}</p>
            <a href="/finished-annotating/" class="back-link">‚Üê Back to Analysis</a>
        `;
        
        // Update error ID color in CSS
        const style = document.createElement('style');
        style.textContent = `
            .error-id { color: ${category.color} !important; }
            .no-errors a { color: ${category.color} !important; }
        `;
        document.head.appendChild(style);
    }
    
    displayErrors() {
        // Find all errors in this category
        const categoryErrors = [];
        
        Object.entries(this.annotations).forEach(([entryId, tags]) => {
            if (tags.includes(this.currentCategory.name)) {
                // Find the corresponding CSV entry
                const csvEntry = this.csvData.find(entry => 
                    (entry.task_id && entry.task_id === entryId) || 
                    entryId.includes(entry.task_id)
                );
                
                if (csvEntry) {
                    categoryErrors.push({
                        id: entryId,
                        entry: csvEntry,
                        tags: tags
                    });
                }
            }
        });
        
        // Update error count
        document.getElementById('errorCount').textContent = `${categoryErrors.length} errors found`;
        
        // Display errors
        if (categoryErrors.length === 0) {
            this.showNoErrors();
        } else {
            this.showErrorList(categoryErrors);
        }
    }
    
    showErrorList(errors) {
        const container = document.getElementById('errorListContent');
        
        let html = '<ul class="error-list">';
        
        errors.forEach((error, index) => {
            const description = error.entry.text || error.entry.task_description || 'No description available';
            const originalResult = error.entry.original_test_result || 'No result';
            const refactoredResult = error.entry.refactored_test_result || 'No result';
            
            html += `
                <li class="error-item">
                    <div class="error-id">Error #${index + 1}: ${error.id}</div>
                    <div class="error-description">${description}</div>
                    <div class="error-meta">
                        <span>üìù Original: ${originalResult}</span>
                        <span>üîß Refactored: ${refactoredResult}</span>
                        <span>üè∑Ô∏è Tags: ${error.tags.join(', ')}</span>
                    </div>
                </li>
            `;
        });
        
        html += '</ul>';
        container.innerHTML = html;
    }
    
    showNoErrors() {
        const container = document.getElementById('errorListContent');
        container.innerHTML = `
            <div class="no-errors">
                <h3>üéâ No Errors Found!</h3>
                <p>There are no annotated errors in the "${this.currentCategory.name}" category.</p>
                <p>This could mean either no errors of this type occurred, or they haven't been annotated yet.</p>
            </div>
        `;
    }
    
    showInvalidCategory() {
        document.getElementById('loadingSpinner').style.display = 'none';
        const container = document.getElementById('categoryContent');
        container.style.display = 'block';
        container.innerHTML = `
            <div class="no-errors">
                <h3>‚ùå Invalid Category</h3>
                <p>The requested error category does not exist. Please select a category from 1-7.</p>
                <div style="margin-top: 15px;">
                    <a href="/finished-annotating/" style="color: #e74c3c; text-decoration: none; font-weight: 500;">‚Üê Return to Analysis</a>
                </div>
            </div>
        `;
    }
    
    showContent() {
        document.getElementById('loadingSpinner').style.display = 'none';
        document.getElementById('categoryContent').style.display = 'block';
    }
}

// Initialize when page loads
document.addEventListener('DOMContentLoaded', () => {
    new DynamicErrorCategoryViewer();
});
</script>
