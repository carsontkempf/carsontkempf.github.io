<script>
class PerformanceHistoryMemory {
    constructor() {
        this.memorySlots = [];
        this.currentMemoryIndex = 0;
        this.maxMemorySlots = 10;
        
        this.initializeElements();
        this.loadMemoryState();
        this.setupEventListeners();
    }

    initializeElements() {
        this.memoryGrid = document.getElementById('memoryGrid');
        this.prevBtn = document.getElementById('prevBtn');
        this.nextBtn = document.getElementById('nextBtn');
        this.memoryInfo = document.getElementById('memoryInfo');
        this.pdfViewer = document.getElementById('pdfViewer');
        this.currentMemoryDisplay = document.getElementById('currentMemoryDisplay');
    }

    async loadMemoryState() {
        // Load memory state with actual PDF paths for the website
        this.memorySlots = [
            { id: 0, pdfNumber: 11, type: 'queue', path: '/assets/pdf-memory/queue/queue_0011.pdf', status: 'newest' },
            { id: 1, pdfNumber: 9, type: 'queue', path: '/assets/pdf-memory/queue/queue_0009.pdf', status: 'active' },
            { id: 2, pdfNumber: 8, type: 'queue', path: '/assets/pdf-memory/queue/queue_0008.pdf', status: 'active' },
            { id: 3, pdfNumber: 7, type: 'queue', path: '/assets/pdf-memory/queue/queue_0007.pdf', status: 'active' },
            { id: 4, pdfNumber: 6, type: 'queue', path: '/assets/pdf-memory/queue/queue_0006.pdf', status: 'active' },
            { id: 5, pdfNumber: 5, type: 'queue', path: '/assets/pdf-memory/queue/queue_0005.pdf', status: 'active' },
            { id: 6, pdfNumber: 4, type: 'queue', path: '/assets/pdf-memory/queue/queue_0004.pdf', status: 'active' },
            { id: 7, pdfNumber: 3, type: 'queue', path: '/assets/pdf-memory/queue/queue_0003.pdf', status: 'active' },
            { id: 8, pdfNumber: 2, type: 'queue', path: '/assets/pdf-memory/queue/queue_0002.pdf', status: 'oldest' },
            { id: 9, pdfNumber: 10, type: 'milestone', path: '/assets/pdf-memory/milestones/milestone_0010.pdf', status: 'milestone' }
        ];

        this.generateMemoryGrid();
        this.loadMostRecent();
    }

    generateMemoryGrid() {
        this.memoryGrid.innerHTML = '';
        
        for (let i = 0; i < this.maxMemorySlots; i++) {
            const slot = document.createElement('div');
            slot.className = 'memory-slot';
            slot.dataset.memoryIndex = i;
            
            const memoryData = this.memorySlots[i];
            
            if (memoryData) {
                slot.textContent = `M[${i}]`;
                slot.title = `Memory[${i}] - PDF #${memoryData.pdfNumber} (${memoryData.type})`;
                
                if (memoryData.type === 'milestone') {
                    slot.classList.add('milestone');
                }
                
                if (i === 0) {
                    slot.classList.add('active');
                }
                
                slot.addEventListener('click', () => this.selectMemorySlot(i));
            } else {
                slot.textContent = `M[${i}]`;
                slot.classList.add('empty');
                slot.title = `Memory[${i}] - Empty`;
            }
            
            this.memoryGrid.appendChild(slot);
        }
    }

    selectMemorySlot(index) {
        const memoryData = this.memorySlots[index];
        if (!memoryData) return;

        this.currentMemoryIndex = index;
        this.updateActiveSlot();
        this.loadPDF(memoryData);
        this.updateNavigationButtons();
        this.updateMemoryDisplay();
    }

    updateActiveSlot() {
        document.querySelectorAll('.memory-slot').forEach(slot => {
            slot.classList.remove('active');
        });
        
        const activeSlot = document.querySelector(`[data-memory-index="${this.currentMemoryIndex}"]`);
        if (activeSlot) {
            activeSlot.classList.add('active');
        }
    }

    loadMostRecent() {
        // Load Memory[0] - most recent performance history
        this.selectMemorySlot(0);
    }

    loadPDF(memoryData) {
        const statusDescriptions = {
            'newest': 'Most Recent (Newest in Queue)',
            'oldest': 'About to be Removed (Oldest in Queue)', 
            'milestone': 'Permanent Milestone',
            'active': 'Active in Queue'
        };

        this.memoryInfo.textContent = `PDF #${memoryData.pdfNumber} - ${statusDescriptions[memoryData.status]}`;
        
        // Load the PDF
        this.pdfViewer.innerHTML = `
            <iframe class="pdf-viewer" 
                    src="${memoryData.path}" 
                    type="application/pdf">
                <p>Your browser does not support PDF viewing. 
                   <a href="${memoryData.path}" target="_blank">Click here to download PDF #${memoryData.pdfNumber}</a>
                </p>
            </iframe>
        `;
    }

    updateNavigationButtons() {
        // Find previous available memory slot
        let prevIndex = -1;
        for (let i = this.currentMemoryIndex - 1; i >= 0; i--) {
            if (this.memorySlots[i]) {
                prevIndex = i;
                break;
            }
        }

        // Find next available memory slot
        let nextIndex = -1;
        for (let i = this.currentMemoryIndex + 1; i < this.maxMemorySlots; i++) {
            if (this.memorySlots[i]) {
                nextIndex = i;
                break;
            }
        }

        this.prevBtn.disabled = prevIndex === -1;
        this.nextBtn.disabled = nextIndex === -1;
    }

    updateMemoryDisplay() {
        const memoryData = this.memorySlots[this.currentMemoryIndex];
        if (memoryData) {
            const statusText = memoryData.status === 'newest' ? 'Most Recent' : 
                             memoryData.status === 'oldest' ? 'About to be Removed' :
                             memoryData.status === 'milestone' ? 'Milestone' : 'Active';
            this.currentMemoryDisplay.textContent = `Memory[${this.currentMemoryIndex}] - ${statusText}`;
        }
    }

    navigateMemory(direction) {
        const currentIndex = this.currentMemoryIndex;
        let targetIndex = -1;

        if (direction === -1) {
            // Previous
            for (let i = currentIndex - 1; i >= 0; i--) {
                if (this.memorySlots[i]) {
                    targetIndex = i;
                    break;
                }
            }
        } else {
            // Next
            for (let i = currentIndex + 1; i < this.maxMemorySlots; i++) {
                if (this.memorySlots[i]) {
                    targetIndex = i;
                    break;
                }
            }
        }

        if (targetIndex !== -1) {
            this.selectMemorySlot(targetIndex);
        }
    }

    setupEventListeners() {
        this.prevBtn.addEventListener('click', () => {
            this.navigateMemory(-1);
        });

        this.nextBtn.addEventListener('click', () => {
            this.navigateMemory(1);
        });

        // Keyboard navigation
        document.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowLeft' && !this.prevBtn.disabled) {
                this.navigateMemory(-1);
            } else if (e.key === 'ArrowRight' && !this.nextBtn.disabled) {
                this.navigateMemory(1);
            } else if (e.key >= '0' && e.key <= '9') {
                const index = parseInt(e.key);
                if (this.memorySlots[index]) {
                    this.selectMemorySlot(index);
                }
            }
        });
    }
}

// Initialize the performance history memory interface when the page loads
document.addEventListener('DOMContentLoaded', () => {
    new PerformanceHistoryMemory();
});
</script>
